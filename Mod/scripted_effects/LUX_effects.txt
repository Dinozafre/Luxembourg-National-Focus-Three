LUX_demand_decolonisation_effect = {
	every_possible_country = {
		limit = {
			exists = no
			any_state = {
				is_core_of = PREV
				has_resistance = yes
				NOT = {
					is_owned_and_controlled_by = PREV
				}
				is_owned_and_controlled_by = ROOT
			}
		}
		ROOT = {
			release_on_controlled = PREV
		}
		set_country_flag = has_been_decolonized_flag
		add_to_array = {
			array = FROM.decolonized_contries
			value = THIS
		}
	}
	every_subject_country = {
		display_individual_scopes = yes
		set_country_flag = has_been_decolonized_flag
		add_to_array = {
			array = FROM.decolonized_contries
			value = THIS
		}
		country_event = {
			id = LUX_diplomacy.119
			hours = 6
			random_hours = 12
		}
		ROOT = {
            set_autonomy = {
				target = PREV
				autonomy_state = autonomy_free
			}
		}
	}
	hidden_effect = {
		every_possible_country = {
			limit = {
				exists = yes
				any_core_state = {
					NOT = {
						is_owned_and_controlled_by = PREV
					}
					has_resistance = yes
				}
			}
			ROOT = {
				every_controlled_state = {
					limit = {
						is_core_of = PREV.PREV
						has_resistance = yes
						is_owned_and_controlled_by = ROOT
					}
					transfer_state_to = PREV.PREV
				}
			}
		}
	}
}
